syncTable() { const tableBody = document.querySelector('#serviciosTable tbody'); tableBody.innerHTML = ''; let totalCosto = 0; this.servicios.forEach((serv) => { totalCosto += (serv.cantidad * serv.costo); const row = ` ${serv.codigo} ${serv.descripcion} ${serv.cantidad} ${serv.costo} `; tableBody.innerHTML += row; }); // Mostrar el total const totalRow = `Total${totalCosto}`; tableBody.innerHTML += totalRow; } addServicio(codigo, descripcion, cantidad, costo) { const servicioExistente = this.servicios.findIndex(serv => serv.codigo === codigo); if (servicioExistente >= 0) { alert('Este servicio ya estÃ¡ agregado. Actualizando cantidad.'); this.servicios[servicioExistente].cantidad += cantidad; } else { this.servicios.push({ codigo, descripcion, cantidad, costo }); } this.syncTable(); } removeServicio(codigo) { this.servicios = this.servicios.filter(serv => serv.codigo !== codigo); this.syncTable(); }
